```bash
npm install --save react-typeme-effect
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-typeme-effect'
import 'react-typeme-effect/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
